## Implementation Pattern (React)

```tsx
import React, { useRef, useEffect, useState } from 'react';

interface AnimatedVisualProps {
  trigger?: 'hover' | 'scroll' | 'load';
}

export const CellularRechargeVisual: React.FC<AnimatedVisualProps> = ({ 
  trigger = 'scroll' 
}) => {
  const svgRef = useRef<SVGSVGElement>(null);
  const [isPlaying, setIsPlaying] = useState(false);
  
  useEffect(() => {
    if (trigger === 'scroll' && svgRef.current) {
      const observer = new IntersectionObserver(
        ([entry]) => {
          if (entry.isIntersecting && !isPlaying) {
            setIsPlaying(true);
          }
        },
        { threshold: 0.5 }
      );
      observer.observe(svgRef.current);
      return () => observer.disconnect();
    }
  }, [trigger, isPlaying]);
  
  return (
    <svg 
      ref={svgRef} 
      className={`w-full h-auto ${isPlaying ? 'animate' : ''}`}
      viewBox="0 0 400 300"
      role="img" 
      aria-label="Cellular energy restoration animation showing a depleted cell being energized by electromagnetic pulses"
    >
      {/* SVG content with CSS animations controlled by .animate class */}
    </svg>
  );
};
```

## Accessibility Checklist

- [ ] `role="img"` + descriptive `aria-label` on all SVGs
- [ ] Decorative elements have `aria-hidden="true"`
- [ ] `tabindex="0"` on interactive elements
- [ ] `@media (prefers-reduced-motion: reduce)` fallback
- [ ] Focus states visible for keyboard navigation
- [ ] Touch targets ≥44px on mobile
- [ ] No flashing faster than 3 times per second (photosensitivity safe)
